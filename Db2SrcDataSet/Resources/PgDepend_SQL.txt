select d.deptype,
  d.classid,
  d.objid, d.objsubid,
  d.refclassid,
  d.refobjid, d.refobjsubid
from pg_depend d
  join pg_class c on (d.classid = c.oid)
  join pg_class rc on (d.refclassid = rc.oid)
where (d.deptype = 'e' -- pg_extension‚ÌˆË‘¶ŠÖŒW objid:pg_proc‘¼ refobjid:pg_extension
    or (d.deptype = 'n' -- pg_trigger‚Æpg_proc‚ÌˆË‘¶ŠÖŒW objid:pg_trigger refobjid:pg_proc
      and d.classid = :pg_trigger_oid
      and d.refclassid = :pg_proc_oid
    )
  )
order by d.objid, d.objsubid, d.refobjid, d.refobjsubid
